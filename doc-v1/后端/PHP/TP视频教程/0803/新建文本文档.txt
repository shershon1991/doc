#创建后台管理员表
create table ci_manager(
	mg_id smallint unsigned not null auto_increment primary key,
	mg_name varchar(30) not null default '',
	mg_pwd char(32) not null default '',
	mg_time int unsigned not null default 0,
	mg_role_id smallint unsigned not null default 0
)engine=MyIsam charset=utf8;
